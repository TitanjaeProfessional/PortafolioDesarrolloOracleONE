¿Cuántos años de diferencia tienes con tu hermano? Escribe un programa que muestre el mensaje ¨Nuestra diferencia de edad es¨, concatenando el resultado de la diferencia de tu edad con la de tu hermano (o de un amigo). La respuesta puede dar negativa, sin duda. No olvides de usar las funciones saltarLinea e imprimir y de incluir la etiqueta meta para resolver problemas de acentuación.



Codigo Implementado

function Ejecucion(){
var nombre1 = prompt("Ingresa el nombre del hermano 1");
var edad1 = prompt("Ingresa la edad del hermano 1");
var nombre2 = prompt("Ingresa el nombre del hermano 2");
var edad2 = prompt("Ingresa la edad del hermano 2");
let difedades = DiferenciaEdadesActualizadas(edad1,edad2);
imprimirSalida("La diferencia de edad entre "+ nombre1 +" y " + nombre2 + " es de : " + difedades);
}
function imprimirSalida(Salida){
document.write(Salida);
}
function DiferenciaEdadesActualizadas(EdadActual1,EdadActual2)
{
var DiferenciaEdades =( EdadActual1 - EdadActual2);
return DiferenciaEdades;
}
Ejecucion();
